LOAD EMPLOYEE
RR <- SELECT Salary >= 30000 FROM EMPLOYEE
Q1 <- PROJECT Ssn, Salary FROM RR

j <- CROSS EMPLOYEE,EMPLOYEE
J <- SELECT EMPLOYEE1_Bdate == EMPLOYEE2_Bdate FROM j
Q <- SELECT EMPLOYEE1_Ssn != EMPLOYEE2_Ssn FROM J
QQ <- SELECT EMPLOYEE1_Super_ssn == EMPLOYEE2_Ssn FROM Q
Q2 <- PROJECT EMPLOYEE1_Ssn, EMPLOYEE1_Super_ssn, EMPLOYEE1_Bdate FROM QQ
RENAME EMPLOYEE1_Ssn TO Ssn FROM Q2
RENAME EMPLOYEE1_Super_ssn TO Super_ssn FROM Q2
RENAME EMPLOYEE1_Bdate TO Bdate FROM Q2

LOAD WORKS_ON
CROSS <- CROSS EMPLOYEE WORKS_ON
CROSS_1 <- SELECT Essn == Ssn FROM CROSS
CROSS_2 <- CROSS CROSS_1,CROSS_1
CROSS_3 <- SELECT CROSS_11_Pno == CROSS_12_Pno FROM CROSS_2
TT <- SELECT CROSS_11_Sex != CROSS_12_Sex FROM CROSS_3
Q3 <- PROJECT CROSS_11_Pno FROM TT
RENAME CROSS_11_Pno TO Pno FROM Q3

CLEAR CROSS
CLEAR CROSS_1
CLEAR CROSS_2
CLEAR CROSS_3
CROSS <- CROSS EMPLOYEE WORKS_ON
CROSS_1 <- SELECT Essn == Ssn FROM CROSS
CROSS_2 <- CROSS CROSS_1,CROSS_1
CROSS_3 <- SELECT CROSS_11_Super_ssn == CROSS_12_Ssn FROM CROSS_2
CROSS_4 <- SELECT CROSS_11_Pno == CROSS_12_Pno FROM CROSS_3
Q4 <- PROJECT CROSS_11_Ssn, CROSS_11_Super_ssn, CROSS_11_Pno FROM CROSS_4
RENAME CROSS_11_Ssn TO Ssn FROM Q4
RENAME CROSS_11_Super_ssn TO Super_ssn FROM Q4
RENAME CROSS_11_Pno TO Pno FROM Q4

LOAD PROJECT
R1 <- SELECT Dnum < 6 FROM PROJECT
R2 <- SELECT Dnum != 2 FROM R1
R3 <- SELECT Dnum != 3 FROM R2
Q5 <- SELECT Dnum != 4 FROM R3

EXPORT Q1
EXPORT Q2
EXPORT Q3
EXPORT Q4
EXPORT Q5
QUIT
